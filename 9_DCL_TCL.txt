# FROM ROOT USER TERMINAL

1. login as the root user

-> mysql -u root -p

2. create new user

-> create user 'monkey'@'localhost' identified by '1234';

then exit from the root user

-> exit


# LOGIN AS NEW USER IN A NEW TERMINAL

-> mysql -u monkey -p
(enter password: 1234)

create a database

mysql> create database mydatabase;

error aayega, now

-> exit;


# LOGIN AS ROOT IN THE MAIN TERMINAL

-> mysql -u root -p

mysql> grant create on *.* to 'monkey'@'localhost';

mysql> exit;
Bye


# LOGIN IN THE NEXT TERMINAL AS USER

-> mysql -u monkey -p

mysql> create database mydatabase;
Query OK, 1 row affected (0.135 sec)

Now create a table

mysql> use mydatabase;
Database changed
mysql> create table students(Sid int, Sname varchar(30), address varchar(50), branch varchar(20));
Query OK, 0 rows affected (0.484 sec)

mysql> insert into students value(1, 'Monkey', 'Gangtok', 'CS');
ERROR 1142 (42000): INSERT command denied to user 'monkey'@'localhost' for table 'students'

-> exit;

# LOGIN AS ROOT AND GRANT ALL PERMISSION

mysql> grant all on *.* to 'monkey'@'localhost';

mysql> exit;
Bye
